            const end = () => d.isDragging = false;
            el.addEventListener('mousedown', start); window.addEventListener('mousemove', move); window.addEventListener('mouseup', end);
            el.addEventListener('touchstart', start); window.addEventListener('touchmove', move); window.addEventListener('touchend', end);
            balls.push(d);
        }

        function phys() {
            const hH = hero.offsetHeight;
            const gX = window.innerWidth / 2;
            balls.forEach(b => {
                if (!b.isDragging) {
                    b.vy += 0.4; b.x += b.vx; b.y += b.vy;
                    if (b.x <= 0 || b.x >= window.innerWidth - b.size) b.vx *= -0.7;
                    if (b.y >= hH - b.size) { b.y = hH - b.size; b.vy *= -0.5; b.vx *= 0.9; }
                    if (b.y <= 0) { b.y = 0; b.vy *= -0.7; }
                    // Goal detect
                    if (b.y < 95 && b.x > gX - 90 && b.x < gX + 90) {
                        score++; sVal.textContent = score;
                        // Play Sound
                        gSound.currentTime = 0;
                        gSound.play();
                        // Reset Ball
                        b.y = hH - 120; b.x = Math.random() * (window.innerWidth - 100) + 50;
                        b.vx = 0; b.vy = 0;
                    }
                }
                b.el.style.transform = `translate3d(${b.x}px, ${b.y}px, 0) rotate(${b.x * 2}deg)`;
            });
            requestAnimationFrame(phys);
        }
        addBall(); phys();

        // Swiper with Pagination Below and airy transition
        new Swiper('.mySwiper', {
            loop: true, 
            slidesPerView: 1, 
            spaceBetween: 30, 
            autoplay: { delay: 3500 },
            effect: 'creative',
            creativeEffect: {
                prev: { shadow: false, translate: ["-120%", 0, -500], opacity: 0 },
                next: { shadow: false, translate: ["120%", 0, -500], opacity: 0 },
            },
            pagination: { el: '.swiper-pagination', clickable: true },
            breakpoints: { 768: { slidesPerView: 1.5 } }
        });
